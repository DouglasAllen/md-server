Quickstart

# Gemfile
source "http://rubygems.org"
ruby '2.3.1'
gem 'rack', '~> 1.5.2'
$>  bundle install
# request_controller.rb
class RequestController
  def call(env)
    [200, {}, ["Hello World"]]
  end
end
# config.ru
require 'rack'
load './request_controller.rb'

Rack::Handler::WEBrick.run(RequestController.new, :Port => 9000)
$>  rackup
